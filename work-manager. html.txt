<script>
  let tasks = [], minutes = [], taskId = 0, minId = 0;

  function load() {
    try { const d = JSON.parse(localStorage.getItem('wm-tasks') || '{}'); tasks = d.tasks || []; taskId = d.nextId || 0; } catch(e) {}
    try { const d = JSON.parse(localStorage.getItem('wm-minutes') || '{}'); minutes = d.minutes || []; minId = d.nextId || 0; } catch(e) {}
    renderTasks(); renderMinutes();
  }

  function saveTasks() { localStorage.setItem('wm-tasks', JSON.stringify({tasks, nextId: taskId})); }
  function saveMinutes() { localStorage.setItem('wm-minutes', JSON.stringify({minutes, nextId: minId})); }

  function showToast(msg) {
    const t = document.getElementById('toast');
    t.textContent = msg; t.classList.add('show');
    setTimeout(() => t.classList.remove('show'), 2000);
  }

  function switchTab(tab, btn) {
    document.querySelectorAll('.tab').forEach(t => t.classList.remove('active'));
    document.querySelectorAll('.section').forEach(s => s.classList.remove('active'));
    btn.classList.add('active');
    document.getElementById(tab + '-section').classList.add('active');
  }

  function addTask(title, priority, date, status) {
    title = title || document.getElementById('task-title').value.trim();
    if (!title) return;
    priority = priority || document.getElementById('task-priority').value;
    date = date !== undefined ? date : document.getElementById('task-date').value;
    status = status || 'pendiente';
    tasks.push({ id: ++taskId, title, priority, date, status });
    saveTasks();
    document.getElementById('task-title').value = '';
    document.getElementById('task-date').value = '';
    renderTasks(); showToast('Tarea agregada');
  }

  function moveTask(id, dir) {
    const order = ['pendiente', 'progreso', 'completada'];
    const t = tasks.find(t => t.id === id); if (!t) return;
    const i = order.indexOf(t.status);
    if (dir === 1 && i < 2) t.status = order[i+1];
    if (dir === -1 && i > 0) t.status = order[i-1];
    saveTasks(); renderTasks();
  }

  function deleteTask(id) {
    tasks = tasks.filter(t => t.id !== id);
    saveTasks(); renderTasks(); showToast('Tarea eliminada');
  }

  function formatDate(d) {
    if (!d) return '';
    const [y,m,day] = d.split('-'); return `${day}/${m}/${y}`;
  }

  function renderTasks() {
    const fs = document.getElementById('filter-status').value;
    const fp = document.getElementById('filter-priority').value;
    ['pendiente','progreso','completada'].forEach(col => {
      const filtered = tasks.filter(t => t.status === col && (fs==='all'||t.status===fs) && (fp==='all'||t.priority===fp));
      document.getElementById('cnt-'+col).textContent = filtered.length;
      const el = document.getElementById('col-'+col);
      if (!filtered.length) { el.innerHTML = '<div class="empty">Sin tareas</div>'; return; }
      el.innerHTML = filtered.map(t => `
        <div class="task-card">
          <div class="task-title">${t.title}</div>
          <div class="task-meta">
            <span class="badge ${t.priority}">${t.priority}</span>
            ${t.date ? `<span class="task-date">${formatDate(t.date)}</span>` : ''}
          </div>
          <div class="task-actions">
            ${col!=='pendiente' ? `<button onclick="moveTask(${t.id},-1)">← Atrás</button>` : ''}
            ${col!=='completada' ? `<button onclick="moveTask(${t.id},1)">Avanzar →</button>` : ''}
            <button class="danger" onclick="deleteTask(${t.id})">Eliminar</button>
          </div>
        </div>`).join('');
    });
  }

  function saveMinute() {
    const title = document.getElementById('min-title').value.trim();
    if (!title) return;
    const date = document.getElementById('min-date').value;
    const participants = document.getElementById('min-participants').value.trim();
    const topics = document.getElementById('min-topics').value.trim();
    const decisions = document.getElementById('min-decisions').value.trim();
    const nextsteps = document.getElementById('min-nextsteps').value.trim().split('\n').filter(s => s.trim());
    minutes.unshift({ id: ++minId, title, date, participants, topics, decisions, nextsteps });
    saveMinutes();
    ['min-title','min-date','min-participants','min-topics','min-decisions','min-nextsteps'].forEach(id => document.getElementById(id).value = '');
    renderMinutes(); showToast('Minuta guardada');
  }

  function renderMinutes() {
    const el = document.getElementById('min-list');
    if (!minutes.length) { el.innerHTML = '<div class="empty" style="margin-bottom:1rem">Sin minutas guardadas</div>'; return; }
    el.innerHTML = minutes.map(m => `
      <div class="min-card" onclick="showMinute(${m.id})">
        <div class="min-card-header">
          <span class="min-card-title">${m.title}</span>
          <span class="min-card-date">${m.date ? formatDate(m.date) : ''}</span>
        </div>
        ${m.participants ? `<div class="min-card-participants">Participantes: ${m.participants}</div>` : ''}
      </div>`).join('');
  }

  function showMinute(id) {
    const m = minutes.find(m => m.id === id); if (!m) return;
    const dv = document.getElementById('min-detail-view');
    dv.innerHTML = `
      <button class="back-btn" onclick="hideMinute()">← Volver</button>
      <div class="min-detail">
        <h3>${m.title}</h3>
        <div class="sub">${m.date ? formatDate(m.date)+' · ' : ''}${m.participants||''}</div>
        ${m.topics ? `<div class="min-section"><div class="min-section-title">Temas tratados</div><div class="min-section-content">${m.topics}</div></div>` : ''}
        ${m.decisions ? `<div class="min-section"><div class="min-section-title">Decisiones tomadas</div><div class="min-section-content">${m.decisions}</div></div>` : ''}
        ${m.nextsteps&&m.nextsteps.length ? `
          <div class="min-section">
            <div class="min-section-title">Próximos pasos</div>
            <ul class="next-steps">
              ${m.nextsteps.map(s=>`
                <li>
                  <span>${s}</span>
                  <button class="to-task-btn" onclick="stepToTask('${encodeURIComponent(s)}')">+ Crear tarea</button>
                </li>`).join('')}
            </ul>
          </div>` : ''}
      </div>`;
    document.getElementById('min-list-view').style.display = 'none';
    dv.style.display = 'block';
  }

  function hideMinute() {
    document.getElementById('min-list-view').style.display = 'block';
    document.getElementById('min-detail-view').style.display = 'none';
  }

  function stepToTask(encoded) {
    const title = decodeURIComponent(encoded);
    tasks.push({ id: ++taskId, title, priority: 'media', date: '', status: 'pendiente' });
    saveTasks();
    document.querySelectorAll('.tab')[0].click();
    renderTasks(); showToast('Tarea creada desde minuta');
  }

  load();
</script>
